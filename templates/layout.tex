% 1) Subsection heading: keep following content with the heading
\makeatletter
\renewcommand{\cvsubsection}[1]{%
  \vspace{\acvSectionContentTopSkip}% global knob below
  \vspace{-3mm}% fine-tune up
  \subsectionstyle{#1}%
  \phantomsection
  \par\nobreak\@afterheading
}
\makeatother

% 2) Global content top skip (single place to control it)
\renewcommand{\acvSectionContentTopSkip}{3mm}

% 3) Bullet lists inside entries: slightly roomier, consistent
\renewenvironment{cvitems}{%
  \vspace{-3.0mm}%
  \begin{justify}%
  \begin{itemize}[leftmargin=2ex,nosep]%
    \setlength{\parskip}{0pt}%
    \setlength{\itemsep}{0.3ex}% space between bullets
    \setlength{\topsep}{0.4ex}% space above/below the list
    \renewcommand{\labelitemi}{\bullet}%
}{%
  \end{itemize}%
  \end{justify}%
  \vspace{1.5mm}%
}

% --- Compact 2-row skill block: full-width header + 2 cols (30/70) ---
\newcommand{\cvskillgroup}[3]{%
  % local scope: help line-breaking so long lists don't overflow
  \begingroup\sloppy\emergencystretch=2em
  % Row 1: full-width heading
  \noindent\begin{tabular*}{\linewidth}{@{\extracolsep{\fill}} p{\linewidth}}
    \skilltypestyle{#1}\\
  \end{tabular*}%
  % small gap
  \vspace{0.2ex}

  % Row 2: two columns (L 30% / R 70%)
  \noindent\begin{tabular*}{\linewidth}{@{\extracolsep{\fill}}%
    >{\raggedright\arraybackslash}p{0.30\linewidth}%
    >{\raggedright\arraybackslash}p{0.70\linewidth}}
    \skilltypestyle{#2} & \skillsetstyle{#3}\\
  \end{tabular*}%
  \par\endgroup
}
% One table per parent, stacked vertically
\makeatletter
\newcommand{\SkillGroupTopSep}{0.8\baselineskip} % space before each parent block
\newcommand{\SkillRowSep}{0.35ex}                % space between rows

% A two-column skills table that wraps nicely
\newenvironment{mycvskills}{%
  \vspace{0.25\baselineskip}%
  \begingroup
    \sloppy\emergencystretch=2em % prevent overflow
    \setlength{\tabcolsep}{0pt}%
    \setlength{\extrarowheight}{0pt}%
    \begin{tabular*}{\linewidth}{@{\extracolsep{\fill}}%
      >{\raggedleft\arraybackslash}p{0.30\linewidth}%
      >{\raggedright\arraybackslash}p{0.67\linewidth}}%
}{%
    \end{tabular*}%
  \endgroup
}

% Row macro: child label + wrapped values
\newcommand{\mycvskill}[2]{%
  \skilltypestyle{#1} & \skillsetstyle{#2} \\[-\SkillRowSep]
}
